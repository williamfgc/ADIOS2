function(add_file_test test_name types)

  foreach(engine_type IN LISTS types)
    message("engine_type: " ${engine_type})
    gtest_add_tests(TARGET ${test_name} ${extra_test_args} 
                    WORKING_DIRECTORY ${${engine_type}_DIR} 
                    EXTRA_ARGS ${engine_type} TEST_SUFFIX _${engine_type})
  endforeach()
endfunction()

set(engine_types BP3 BP4)
if (ADIOS2_HAVE_HDF5)
  list(APPEND engine_types HDF5)
endif()

add_file_test(TestWriteReadGlobalArrays "${engine_types}")
